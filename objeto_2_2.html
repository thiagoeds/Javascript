Uma Array é uma listagem de coisas, ou seja, ela esta toda numerada, lembrando que ele se inicia a contagem do numero (0).
Outras dicas importantes:
Como identificar uma Array, ela vai iniciar com [colchetes] e objetos iniciam com {chaves}.

let carro = {
nome:'Toyota',
modelo:'Corolla',
peso:'1860kg',
ligado:false,
ligar:function(){
this.ligado = true;
console.log('vruum, vruuuuum!');

},

acelerar:function() {
if(this.ligado == true) {
console.log('Riiiihhhiiisssshhh')
} else {
console.log(this.nome+" "+this.modelo+" não está ligado!");
}
}
};

console.log( 'Modelo: '+carro.modelo );

carro.ligar();
carro.acelerar();
